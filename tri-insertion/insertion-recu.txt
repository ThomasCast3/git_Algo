Si index est inf a la taille du tableau

    static void permuter(int[] tab, int index) {
        int perm;
        if ( index  != 0 && tab[index - 1] > tab[index]) {
            perm = tab[index - 1];
            tab[index - 1] = tab[index];
            tab[index] = perm;
            permuter(tab, index -= 1);
        }

    }
static void ranger(int[] tab, int i) {
        if (i < tab.length) {
            permuter(tab, i);

            ranger(tab, i += 1);
        }

    }
